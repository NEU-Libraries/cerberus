def self.find_by_nuid(nuid) 
  results = nuid_unique_query(nuid) 

  if results.length == 0 
    raise Exceptions::NoSuchNuidError.new(nuid) 
  else
    Employee.safe_employee_lookup(results.first["id"]) 
  end
end
